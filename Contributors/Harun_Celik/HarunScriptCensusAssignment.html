This code is for creating variables to display diversity in the acs data for Illinois

options(tigris_use_cache = TRUE)

race_vars <- c(White = "B02001_002", Black = "B02001_003", American_Indian = "B02001_004", Asian = "B02001_005", Native_Hawaiian = "B02001_006", Other = "B02001_007")

ill_diversity_total <- get_acs(geography ="county",
                           survey = "acs5",
                           state = "17",
                           geometry = TRUE,
                           variables = race_vars,
                           summary_var = c(TotalPop = "B02001_001"))
## Getting data from the 2016-2020 5-year ACS
view(ill_diversity_total)

#Calculating Percentages for diversity values in Illinois

ill_diversity_total_pct <- ill_diversity_total %>%
  mutate(Percentage = 100 * (estimate/summary_est))

view(ill_diversity_total_pct)

#Calculating Percentages for diversity values in South Dakota

sd_diversity_total_pct <- sd_diversity_total %>%
  mutate(Percentage = 100 * (estimate/summary_est))

view(sd_diversity_total_pct)

#Mapping Diversity values by facet for Illinois

ggplot(ill_diversity_total_pct, aes(fill = Percentage, color = Percentage)) + 
  geom_sf() +
  theme_minimal() +
  coord_sf(datum = NA) +
  scale_fill_viridis_c(option = "plasma", alpha = 1, end = .8) +
  scale_color_viridis_c(option = "plasma", alpha = 1, end = .8, guide = FALSE) +
  facet_wrap(~variable) + 
  labs(title = "Diversity of Race in Illinois by County",
       subtitle = "2016 - 2020",
       caption = "Data source: 2016-2020 ACS.\nData acquired with the R tidycensus package.", 
       fill = "Percentage")
## Warning: It is deprecated to specify `guide = FALSE` to remove a guide. Please
## use `guide = "none"` instead.
